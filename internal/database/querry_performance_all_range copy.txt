[
  // 1. Lọc theo done_date
  {
    $match: {
      assignee_id: "lyth@ikameglobal.com",
      done_date: {
        $gte: new Date("2025-07-01T00:00:00.000Z"),
        $lte: new Date("2025-07-30T23:59:59.000Z")
      }
    }
  },

  // 2. Tính week_start từ done_date (mỗi tuần bắt đầu từ thứ 2)
  {
    $addFields: {
      week_start: {
        $dateTrunc: {
          date: "$done_date",
          unit: "week",
          binSize: 1,
          startOfWeek: "monday"
        }
      }
    }
  },

  // 3. Lookup level theo team
  {
    $lookup: {
      from: "level",
      localField: "team",
      foreignField: "team",
      as: "level_info"
    }
  },
  { $unwind: "$level_info" },

  // 4. Tính performance_point từ levelPoint
  {
    $addFields: {
      performance_point: {
        $let: {
          vars: {
            idx: { $subtract: ["$level", 1] },
            arr: "$level_info.levelPoint"
          },
          in: {
            $cond: [
              { $lt: ["$$idx", { $size: "$$arr" }] },
              { $arrayElemAt: ["$$arr", "$$idx"] },
              "$level"
            ]
          }
        }
      }
    }
  },

  // 5. Lookup creative-tool theo tool
  {
    $lookup: {
      from: "creative-tool",
      localField: "tool",
      foreignField: "tool_name",
      as: "tool_info"
    }
  },

  // 6. Tách tool loại t và q
  {
    $addFields: {
      tool_points_t: {
        $map: {
          input: {
            $filter: {
              input: "$tool_info",
              as: "t",
              cond: { $eq: ["$$t.type", "t"] }
            }
          },
          as: "t",
          in: {
            tool_name: "$$t.tool_name",
            point: {
              $let: {
                vars: {
                  idx: { $subtract: ["$level", 1] },
                  arr: "$$t.point"
                },
                in: {
                  $cond: [
                    { $lt: ["$$idx", { $size: "$$arr" }] },
                    { $arrayElemAt: ["$$arr", "$$idx"] },
                    { $arrayElemAt: ["$$arr", 0] }
                  ]
                }
              }
            }
          }
        }
      },
      tool_points_q: {
        $map: {
          input: {
            $filter: {
              input: "$tool_info",
              as: "t",
              cond: { $eq: ["$$t.type", "q"] }
            }
          },
          as: "t",
          in: {
            tool_name: "$$t.tool_name",
            point: { $arrayElemAt: ["$$t.point", 0] }
          }
        }
      }
    }
  },

  // 7. Tính tool_factor và creative_process_point
  {
    $addFields: {
      tool_factor: {
        $cond: {
          if: { $eq: [{ $size: "$tool_points_t" }, 0] },
          then: 0,
          else: {
            $reduce: {
              input: {
                $map: {
                  input: "$tool_points_t",
                  as: "tp",
                  in: {
                    $cond: [{ $lt: ["$$tp.point", 1] }, "$$tp.point", 1]
                  }
                }
              },
              initialValue: 1,
              in: { $multiply: ["$$value", "$$this"] }
            }
          }
        }
      },
      creative_process_point: {
        $sum: {
          $map: {
            input: "$tool_points_q",
            as: "tp",
            in: "$$tp.point"
          }
        }
      }
    }
  },

  // 8. Tính creative_task_point
  {
    $addFields: {
      creative_task_point: {
        $multiply: ["$performance_point", "$tool_factor"]
      }
    }
  },

  // 9. Tính base_point
  {
    $addFields: {
      base_point: {
        $add: [
          { $subtract: ["$performance_point", "$creative_task_point"] },
          "$creative_process_point"
        ]
      }
    }
  },

  // 10. Project các field cần thiết
  {
    $project: {
      task_name: 1,
      assignee_id: 1,
      team: 1,
      level: 1,
      performance_point: 1,
      tool_points_t: 1,
      tool_points_q: 1,
      tool_factor: 1,
      creative_process_point: 1,
      creative_task_point: 1,
      base_point: 1,
      done_date: 1,
      week_start: 1
    }
  },

  // 11. Group theo tuần
  {
    $group: {
      _id: "$week_start",
      total_performance_point: { $sum: "$performance_point" },
      total_creative_process_point: { $sum: "$creative_process_point" },
      total_creative_task_point: { $sum: "$creative_task_point" },
      total_base_point: { $sum: "$base_point" },
      tasks: { $push: "$$ROOT" }
    }
  },

  // 12. Sort theo tuần
  { $sort: { _id: 1 } }
]
